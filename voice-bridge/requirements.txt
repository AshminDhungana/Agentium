# Agentium Voice Bridge — host-side Python dependencies
# Install inside the venv created by install-voice-bridge.sh:
#   pip install -r requirements.txt

# WebSocket server (browser sync)
websockets==16.0

# Speech Recognition (Google STT + mic capture)
SpeechRecognition==3.14.5
PyAudio==0.2.14          # mic hardware access

# Offline STT fallback — used automatically by main.py when Google's free
# Web Speech API returns a RequestError (it is rate-limited and not intended
# for commercial/production use). Download a model and set VOSK_MODEL_PATH
# in env.conf to enable; if no model is found the bridge logs once and
# continues with Google as the sole engine (no behavior change without a
# model present).
vosk==0.3.45

# TTS
pyttsx3==2.99

# ── Jarvis upgrade (Phases A–H) ──────────────────────────────────────────────
# Wake-word detection (streaming, ~ms latency). Apache-licensed; free unlike
# Porcupine. Optional: if the model isn't present the bridge falls back to
# REQUIRE_WAKE_WORD=false behavior. Bundle a .onnx model and set WAKE_WORD_MODEL.
openwakeword

# Speech/silence detection (Silero VAD). MIT-licensed, ~2MB ONNX, sub-ms/frame.
silero-vad
onnxruntime

# HTTP client for backend API calls (voice config, speaker ID, STT relay).
aiohttp

# Local neural TTS (Kokoro-82M). Apache 2.0, ~327MB weights, real-time on CPU.
# Pulls in torch. Plays via sounddevice; soundfile writes the WAV chunks.
kokoro
soundfile
sounddevice

# Acoustic echo cancellation for barge-in on open speakers (Phase C).
webrtc-audio-processing

# Optional noise suppression ahead of VAD for HVAC/appliance noise (Phase H).
noisereduce

# JWT verification (same library used by the backend)
python-jose[cryptography]==3.5.0

# Requests (fallback HTTP — stdlib urllib is used by default)
requests==2.32.5
